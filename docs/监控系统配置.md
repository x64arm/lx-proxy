## 🔧 安装配置

### 1. Docker Compose 配置

**文件：** `docker-compose.monitoring.yml`

```yaml
version: '3.8'

services:
  # Prometheus
  prometheus:
    image: prom/prometheus:latest
    container_name: prometheus
    ports:
      - "9090:9090"
    volumes:
      - ./monitoring/prometheus.yml:/etc/prometheus/prometheus.yml
      - prometheus_data:/prometheus
    command:
      - '--config.file=/etc/prometheus/prometheus.yml'
      - '--storage.tsdb.path=/prometheus'
      - '--web.console.libraries=/etc/prometheus/console_libraries'
      - '--web.console.templates=/etc/prometheus/consoles'
      - '--storage.tsdb.retention.time=200h'
      - '--web.enable-lifecycle'
    restart: unless-stopped
    networks:
      - monitoring

  # Grafana
  grafana:
    image: grafana/grafana:latest
    container_name: grafana
    ports:
      - "3000:3000"
    environment:
      - GF_SECURITY_ADMIN_USER=admin
      - GF_SECURITY_ADMIN_PASSWORD=grafana_admin
      - GF_USERS_ALLOW_SIGN_UP=false
    volumes:
      - grafana_data:/var/lib/grafana
      - ./monitoring/grafana/dashboards:/etc/grafana/dashboards
      - ./monitoring/grafana/provisioning:/etc/grafana/provisioning
    restart: unless-stopped
    networks:
      - monitoring
    depends_on:
      - prometheus

  # Node Exporter (系统监控)
  node_exporter:
    image: quay.io/prometheus/node-exporter:latest
    container_name: node_exporter
    ports:
      - "9100:9100"
    volumes:
      - /proc:/host/proc:ro
      - /sys:/host/sys:ro
      - /:/rootfs:ro
    command:
      - '--path.procfs=/host/proc'
      - '--path.sysfs=/host/sys'
      - '--path.rootfs=/rootfs'
      - '--collector.filesystem.mount-points-exclude=^/(sys|proc|dev|host|etc)($$|/)'
    restart: unless-stopped
    networks:
      - monitoring

  # Postgres Exporter (数据库监控)
  postgres_exporter:
    image: quay.io/prometheuscommunity/postgres-exporter:latest
    container_name: postgres_exporter
    ports:
      - "9187:9187"
    environment:
      - DATA_SOURCE_NAME=postgresql://postgres:postgres@db:5432/lx_proxy?sslmode=disable
    restart: unless-stopped
    networks:
      - monitoring
    depends_on:
      - db

networks:
  monitoring:
    driver: bridge

volumes:
  prometheus_data: {}
  grafana_data: {}
```

```yaml
    static_configs:
      - targets: ['localhost:9090']

  # Node Exporter (系统指标)

    static_configs:
      - targets: ['node_exporter:9100']

  # Postgres Exporter (数据库指标)
  - job_name: 'postgres'
    static_configs:
      - targets: ['postgres_exporter:9187']

  # LX-Proxy Backend (应用指标)
  - job_name: 'lx_proxy_backend'
    metrics_path: '/metrics'
    static_configs:
      - targets: ['backend:8080']
    scrape_interval: 10s
```

## 🎯 Grafana 仪表板

### 1. 系统概览

**ID:** 1860 (Node Exporter Full)

导入现成的 Node Exporter 仪表板，监控：
- CPU、内存、磁盘
- 网络接口
- 系统负载

### 2. LX-Proxy 仪表板

创建自定义仪表板，包含：

**系统状态面板：**
```promql
# CPU 使用率
100 - (avg by(instance) (irate(node_cpu_seconds_total{mode="idle"}[5m])) * 100)

# 内存使用率
(1 - (node_memory_MemAvailable_bytes / node_memory_MemTotal_bytes)) * 100

# 磁盘使用率
(1 - (node_filesystem_avail_bytes / node_filesystem_size_bytes)) * 100
```

**应用指标面板：**
```promql
# 活跃用户数
lx_proxy_users_active

# 启用的入站配置数
lx_proxy_inbounds_enabled

# API 请求速率
rate(lx_proxy_api_requests_total[5m])

# API 响应时间（P95）
histogram_quantile(0.95, rate(lx_proxy_api_request_duration_seconds_bucket[5m]))
```

**数据库面板：**
```promql
# 数据库连接数
pg_stat_activity_count

# 查询速率
rate(pg_stat_statements_calls_total[5m])

# 缓存命中率
pg_stat_database_blks_hit / (pg_stat_database_blks_hit + pg_stat_database_blks_read)
```

## 🚨 告警配置

### Prometheus 告警规则

**文件：** `monitoring/prometheus/rules.yml`

```yaml
groups:
  - name: lx_proxy_alerts
    interval: 30s
    rules:
      # 服务宕机
      - alert: LXProxyDown
        expr: up{job="lx_proxy_backend"} == 0
        for: 1m
        labels:
          severity: critical
        annotations:
          summary: "LX-Proxy 服务宕机"
          description: "LX-Proxy 服务已宕机超过 1 分钟"

      # CPU 过高
      - alert: HighCPUUsage
        expr: 100 - (avg by(instance) (irate(node_cpu_seconds_total{mode="idle"}[5m])) * 100) > 80
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "CPU 使用率过高"
          description: "CPU 使用率超过 80% (当前值：{{ $value }}%)"

      # 内存过高
      - alert: HighMemoryUsage
        expr: (1 - (node_memory_MemAvailable_bytes / node_memory_MemTotal_bytes)) * 100 > 85
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "内存使用率过高"
          description: "内存使用率超过 85% (当前值：{{ $value }}%)"

      # 磁盘空间不足
      - alert: LowDiskSpace
        expr: (1 - (node_filesystem_avail_bytes / node_filesystem_size_bytes)) * 100 > 85
        for: 10m
        labels:
          severity: warning
        annotations:
          summary: "磁盘空间不足"
          description: "磁盘使用率超过 85% (当前值：{{ $value }}%)"

      # 数据库连接过多
      - alert: HighDatabaseConnections
        expr: pg_stat_activity_count > 50
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "数据库连接数过多"
          description: "数据库连接数超过 50 (当前值：{{ $value }})"

      # API 错误率高
      - alert: HighAPIErrorRate
        expr: rate(lx_proxy_api_requests_total{status=~"5.."}[5m]) / rate(lx_proxy_api_requests_total[5m]) > 0.05
        for: 5m
        labels:
          severity: critical
        annotations:
          summary: "API 错误率过高"
          description: "API 错误率超过 5% (当前值：{{ $value | humanizePercentage }})"
```

## 📊 应用指标埋点

### Rust 后端指标

**文件：** `backend/src/metrics.rs`

```rust
use prometheus::{
    register_counter, register_gauge, register_histogram,
    Counter, Gauge, Histogram, Encoder, TextEncoder,
};
use lazy_static::lazy_static;

lazy_static! {
    // 活跃用户数
    pub static ref USERS_ACTIVE: Gauge = register_gauge!(
        "lx_proxy_users_active",
        "Number of active users"
    ).unwrap();

    // 启用的入站配置数
    pub static ref INBOUNDS_ENABLED: Gauge = register_gauge!(
        "lx_proxy_inbounds_enabled",
        "Number of enabled inbounds"
    ).unwrap();

    // API 请求总数
    pub static ref API_REQUESTS_TOTAL: Counter = register_counter!(
        "lx_proxy_api_requests_total",
        "Total number of API requests",
        &["method", "endpoint", "status"]
    ).unwrap();

    // API 响应时间
    pub static ref API_REQUEST_DURATION: Histogram = register_histogram!(
        "lx_proxy_api_request_duration_seconds",
        "API request duration in seconds",
        vec![0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0]
    ).unwrap();

    // 流量统计
    pub static ref TRAFFIC_TOTAL: Counter = register_counter!(
        "lx_proxy_traffic_bytes_total",
        "Total traffic in bytes",
        &["type", "inbound_id"]
    ).unwrap();
}

// 暴露指标端点
pub async fn metrics_handler() -> Result<String, StatusCode> {
    let encoder = TextEncoder::new();
    let metric_families = prometheus::gather();
    
    let mut buffer = Vec::new();
    encoder.encode(&metric_families, &mut buffer)
        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;
    
    String::from_utf8(buffer)
        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)
}
```

## 📝 使用指南

### 启动监控

```bash
# 启动监控服务
docker-compose -f docker-compose.yml -f docker-compose.monitoring.yml up -d

# 查看状态
docker-compose ps

# 访问服务
# Prometheus: http://localhost:9090
# Grafana: http://localhost:3000 (admin/grafana_admin)
```

### 导入仪表板

1. 登录 Grafana
2. 导航到 Dashboards → Import
3. 输入仪表板 ID 或上传 JSON 文件
4. 选择数据源（Prometheus）
5. 点击 Import

### 配置告警

1. 在 Grafana 中导航到 Alerting
2. 创建 Contact Point（邮件/Slack/钉钉）
3. 创建 Notification Policy
4. 为仪表板面板添加告警规则
