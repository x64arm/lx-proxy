# Docker
docker run -i grafana/k6 run -

#### 测试脚本示例
```javascript
// load_test.js
import http from 'k6/http';
import { check, sleep } from 'k6';

export const options = {
  stages: [
    { duration: '30s', target: 100 },   // 热身到 100 用户
    { duration: '1m', target: 100 },    // 保持 100 用户
    { duration: '30s', target: 500 },   // 增加到 500 用户
    { duration: '1m', target: 500 },    // 保持 500 用户
    { duration: '30s', target: 0 },     // 逐渐降低到 0
  ],
  thresholds: {
    http_req_duration: ['p(95)<500'],   // 95% 请求应 < 500ms
    http_req_failed: ['rate<0.01'],     // 错误率 < 1%
  },
};

export default function () {
  const params = {
    headers: {
      'Authorization': `Bearer ${__ENV.TOKEN}`,
    },
  };

  // 测试流量统计 API
  const res1 = http.get('http://localhost:8080/api/traffic/summary', params);
  check(res1, {
    'traffic status is 200': (r) => r.status === 200,
    'traffic latency < 200ms': (r) => r.timings.duration < 200,
  });

  sleep(1);

  // 测试用户列表 API
  const res2 = http.get('http://localhost:8080/api/users', params);
  check(res2, {
    'users status is 200': (r) => r.status === 200,
    'users latency < 300ms': (r) => r.timings.duration < 300,
  });

  sleep(1);
}
```

#### 运行测试

# 输出 JSON 结果
k6 run --out json=results.json load_test.js

# 生成 HTML 报告
k6 run --out html=report.html load_test.js

## 📈 缓存命中率监控

### 1. Redis 缓存监控

#### 关键指标
```bash
# 连接 Redis
redis-cli

# 查看统计信息
INFO stats

# 关键指标
- keyspace_hits: 缓存命中次数
- keyspace_misses: 缓存未命中次数
- hit_rate = keyspace_hits / (keyspace_hits + keyspace_misses)
```

#### 命中率计算脚本
```bash
#!/bin/bash
# cache_hit_rate.sh

REDIS_CLI="redis-cli"

# 获取统计
STATS=$($REDIS_CLI INFO stats)

HITS=$(echo "$STATS" | grep "keyspace_hits:" | cut -d: -f2 | tr -d '\r')
MISSES=$(echo "$STATS" | grep "keyspace_misses:" | cut -d: -f2 | tr -d '\r')

# 计算命中率
if [ -z "$HITS" ] || [ -z "$MISSES" ]; then
    echo "无法获取 Redis 统计"
    exit 1
fi

TOTAL=$((HITS + MISSES))
if [ $TOTAL -eq 0 ]; then
    HIT_RATE="0.00"
else
    HIT_RATE=$(echo "scale=4; $HITS / $TOTAL * 100" | bc)
fi

echo "=== Redis 缓存统计 ==="
echo "命中次数：$HITS"
echo "未命中次数：$MISSES"
echo "命中率：${HIT_RATE}%"
echo ""

# 告警阈值
if (( $(echo "$HIT_RATE < 50" | bc -l) )); then
    echo "⚠️  警告：缓存命中率低于 50%！"
elif (( $(echo "$HIT_RATE < 70" | bc -l) )); then
    echo "⚠️  注意：缓存命中率偏低"
else
    echo "✅ 缓存命中率良好"
fi
```

### 2. 应用层缓存监控

#### Rust 后端缓存统计

```rust
// backend/src/cache.rs 中添加统计功能

use std::sync::atomic::{AtomicU64, Ordering};

static CACHE_HITS: AtomicU64 = AtomicU64::new(0);
static CACHE_MISSES: AtomicU64 = AtomicU64::new(0);

pub fn record_hit() {
    CACHE_HITS.fetch_add(1, Ordering::Relaxed);
}

pub fn record_miss() {
    CACHE_MISSES.fetch_add(1, Ordering::Relaxed);
}

pub fn get_stats() -> CacheStats {
    let hits = CACHE_HITS.load(Ordering::Relaxed);
    let misses = CACHE_MISSES.load(Ordering::Relaxed);
    let total = hits + misses;
    let hit_rate = if total > 0 {
        hits as f64 / total as f64
    } else {
        0.0
    };

    CacheStats {
        hits,
        misses,
        hit_rate,
    }
}

#[derive(serde::Serialize)]
pub struct CacheStats {
    pub hits: u64,
    pub misses: u64,
    pub hit_rate: f64,
}

// 添加 API 端点
// GET /api/cache/stats
pub async fn get_cache_stats() -> Json<CacheStats> {
    Json(get_stats())
}
```

#### API 端点使用
```bash
# 获取缓存统计
curl http://localhost:8080/api/cache/stats \
  -H "Authorization: Bearer YOUR_TOKEN"

# 响应示例
{
  "hits": 15234,
  "misses": 2341,
  "hit_rate": 0.8667
}
```
